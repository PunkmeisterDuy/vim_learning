# Search

## / Search
/ ; to search \
n ; next search\
N ; previous text\
// ; access last search

## ? Search
? ; to search backwards \
n; previous search      \
N ; previous search     \
?? ; last search        \
                        \
## * and # Search
/* to search word currently on forwards \
/# to search word currently on backwards

## Useful Commands
:%s/[search]/[replace]/[options]
### options
g ; global\
i; ignore case\
c ; give confirmation\
gic ; global ignore confirmation\
